package groovy j2ee
import groovy lang groovyobject
import groovy lang groovyshell
import org codehaus groovy runtime invokerhelper
import javax naming context
import javax naming initialcontext
import java util properties
/**
* a j2ee console
*
* @author <a href="mailto:james@coredevelopers.net">james strachan</a>
* @version $revision$
*/
public class j2eeconsole
public static void main string args
if  args length <  0
system out println
return
string home   args
properties p   new properties
system setproperty    home
p put context initial_context_factory
p put
p put    home
if  args length > 1
string conf   args
system setproperty    conf
p put    conf
if  args length > 2
string copy   args
system setproperty    copy
p put    copy
try
initialcontext ctx   new initialcontext p
groovyshell shell   new groovyshell
shell setvariable    ctx
//shell.evaluate("src/test/groovy/j2ee/createdata.groovy");
//shell.evaluate("src/main/groovy/ui/console.groovy");
groovyobject console    groovyobject  invokerhelper invokeconstructorof    null
console setproperty    shell
console invokemethod    null
/*
*/
catch  exception e
system out println     e
