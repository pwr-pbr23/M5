public static char tolowercase char letter  char charset
if  charset    unicoderussian
if  letter >       letter <
return letter
if  letter >       letter <
return  char   letter   32
if  charset    koi8
if  letter >  0xe0    letter <  0xff
return  char   letter   32
if  letter >  0xc0    letter <  0xdf
return letter
if  charset    cp1251
if  letter >  0xc0    letter <  0xdf
return  char   letter   32
if  letter >  0xe0    letter <  0xff
return letter
return character tolowercase letter
