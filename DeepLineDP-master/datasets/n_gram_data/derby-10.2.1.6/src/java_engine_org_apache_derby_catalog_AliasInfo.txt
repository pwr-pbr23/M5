package org apache derby catalog
/**
*
* an interface for describing an alias in cloudscape systems.
*
* in a cloudscape system, an alias can be one of the following:
* <ul>
* <li>method alias
* <li>class alias
* <li>synonym
* <li>user-defined aggregate
* </ul>
*
*/
public interface aliasinfo
/**
* public statics for the various alias types as both char and string.
*/
public static final char alias_type_procedure_as_char
public static final char alias_type_function_as_char
public static final char alias_type_synonym_as_char
public static final string alias_type_procedure_as_string
public static final string alias_type_function_as_string
public static final string alias_type_synonym_as_string
/**
* public statics for the various alias name spaces as both char and string.
*/
public static final char alias_name_space_procedure_as_char
public static final char alias_name_space_function_as_char
public static final char alias_name_space_synonym_as_char
public static final string alias_name_space_procedure_as_string
public static final string alias_name_space_function_as_string
public static final string alias_name_space_synonym_as_string
/**
* get the name of the static method that the alias
* represents at the source database.  (only meaningful for
* method aliases )
*
* @return the name of the static method that the alias
* represents at the source database.
*/
public string getmethodname
