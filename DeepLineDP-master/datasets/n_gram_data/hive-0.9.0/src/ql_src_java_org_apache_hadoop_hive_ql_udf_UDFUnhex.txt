if  s    null

// append a leading 0 if needed
string str
if  s getlength   % 2    1
str       s tostring
else
str   s tostring
byte result   new byte
for  int i   0  i < str length    i    2
try
result     byte  integer parseint str substring i  i   2   16
catch  numberformatexception e
